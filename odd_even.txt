// print all even and odd numbers between 1-100 
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<math.h>

int main()
{
    int i;
    printf("odd numbers are:");
    for(i=1;i<=100;i++)
    {
        if(i%2!=0)
        printf("%d",i);
    }
    printf("\n");
    printf("even numbers are:");
    for(i=1;i<=100;i++)
    {
        if(i%2==0)
        printf("%d",i);
    }
    return 0;
}
